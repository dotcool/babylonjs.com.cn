javascript
BABYLON.SceneLoader.RegisterPlugin({
extensions: “.babylon”,
importMesh: function (meshesNames, scene, data, rootUrl, meshes, particleSystems, skeletons) {
return true;
},
load: function (scene, data, rootUrl) {
return true;
}
});
